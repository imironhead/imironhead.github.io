const int table[32] = {
  0, 1, 1, 2, 1, 2, 2, 3, 1, 2, 2, 3, 2, 3, 3, 4, 1, 2, 2, 3, 2, 3, 3,
  4, 2, 3, 3, 4, 3, 4, 4, 5
};

int hammingWeight(uint32_t n) {
    int c = 0;

    while (n) {
        c += table[n & 31];
        n >>= 5;
    }

    return c;
}

var table = [
  0, 1, 1, 2, 1, 2, 2, 3, 1, 2, 2, 3, 2, 3, 3, 4, 1, 2, 2, 3, 2, 3, 3, 4,
  2, 3, 3, 4, 3, 4, 4, 5
];

var hammingWeight = function(n) {
    var c = 0;

    if (n >= 2147483648) {
        c = 1;
        n -= 2147483648;
    }

    while (n > 0) {
        c += table[n & 31];
        n >>= 5;
    }

    return c;
};
