Find Peak Element

cpp

class Solution {
public:
  int findPeakElement(vector<int>& nums) {
    int i = 0;
    int j = nums.size() - 1;
    int m;

    while (i < j) {
      m = (i + j) / 2;

      if (nums[m] < nums[m + 1]) {
        i = m + 1;
      } else {
        j = m;
      }
    }

    return i;
  }
};
