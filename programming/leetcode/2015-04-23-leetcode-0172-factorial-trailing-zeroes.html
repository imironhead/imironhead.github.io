LeetCode 0172: Factorial Trailing Zeroes

Factorial Trailing Zeroes

cpp

class Solution {
public:
  int trailingZeroes(int n) {
    int c(0);

    while (n) {
      c += n / 5;
      n /= 5;
    }

    return c;
  }
};

javascript

/**
 * @param {number} n
 * @return {number}
 */
var trailingZeroes = function(n) {
  var c = 0;

  while (n > 0) {
    n = Math.floor(n / 5);
    c += n;
  }

  return c;
};
