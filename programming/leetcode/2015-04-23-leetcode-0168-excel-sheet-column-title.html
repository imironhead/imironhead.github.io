LeetCode 0168: Excel Sheet Column Title

Excel Sheet Column Title

cpp

class Solution {
public:
  string convertToTitle(int n) {
    string ans("");

    while (n) {
      ans = static_cast<char>('A' + (n - 1) % 26) + ans;

      n = (n - 1) / 26;
    }

    return ans;
  }
};

javascript

/**
 * @param {number} n
 * @return {string}
 */
var convertToTitle = function(n) {
  var r = '';

  while (n > 0) {
    r = String.fromCharCode(65 + (n - 1) % 26) + r;

    n = Math.floor((n - 1) / 26);
  }

  return r;
};

python

class Solution:
  # @param {integer} n
  # @return {string}
  def convertToTitle(self, n):
    r = ''

    while n > 0 :
      r = chr(ord('A') + (n - 1) % 26) + r

      n = (n - 1) / 26

    return r

  r = ''

  end
